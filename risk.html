max_troops=10+1 # 1 is for the extra index

# packages
import numpy as np

# intialize global variables
outcomes_of_dice_combos = np.zeros((3,2,3))
lost_troops_offender = np.zeros((max_troops,max_troops))
lost_troops_defender = np.zeros((max_troops,max_troops))
victories_for_troop_combos=np.zeros((max_troops,max_troops))

#
class OneRisk:
    def __init__(self,offender_troops,defender_troops):
        self.offender_troops=offender_troops
        self.defender_troops=defender_troops
        self.offender_troops_orig=deepcopy(self.offender_troops)
        self.defender_troops_orig=deepcopy(self.defender_troops)

    def throw_die(self):
        return randint(1,6)

    def attack(self):
        # determining how many dice to throw
        if self.offender_troops>=3:
            nr_of_offender_dice=3
        elif self.offender_troops==2:
            nr_of_offender_dice=2
        elif self.offender_troops==1:
            nr_of_offender_dice=1
        else:
            raise Exception("offender troops not large enough") 

        if self.defender_troops>=2:
            nr_of_defender_dice=2
        elif self.defender_troops==1:
            nr_of_defender_dice=1
        else:
            raise Exception("defence troops not large enough")

        #throwing dice
        offense_dice=sorted([self.throw_die() for _ in range(nr_of_offender_dice)],reverse=True)
        defence_dice=sorted([self.throw_die() for _ in range(nr_of_defender_dice)],reverse=True)
        # if verbose:
        #     print(f"{'offense_dice':<30}{offense_dice}")
        #     print(f"{'defence_dice':<30}{defence_dice}")
        #     print('  ')

        # calculating how many time offender wins
        wins_counter=0
        if offense_dice[0]>defence_dice[0]:
            wins_counter+=1
        if nr_of_offender_dice>1 and nr_of_defender_dice>1:
            if offense_dice[1]>defence_dice[1]:
                wins_counter+=1
        outcomes_of_dice_combos[nr_of_offender_dice-1][nr_of_defender_dice-1][wins_counter]+=1

        # substracting nr of troops based on outcome and nr of dice used
        if wins_counter==2: # if offender wins both, substract two defender troops
            self.defender_troops-=2
        elif wins_counter==0: # if offender loses all fights, substract the number of dice used
            if nr_of_defender_dice==2 and nr_of_offender_dice>=2:
                self.offender_troops-=2
            else:
                self.offender_troops-=1
        elif wins_counter==1: # if offender wins one throw of a die, determine how many dice offender/defender have and substract accordingly
            if nr_of_defender_dice==2 and nr_of_offender_dice>=2: # if two dice matched, substract one offender troop
                self.offender_troops-=1
            self.defender_troops-=1 # subtract one from defender anyways
        #return wins_counter


    def calculate_outcome_of_one_battle(self):
        global lost_troops_offender, lost_troops_defender,victories_for_troop_combos

        # attack as long as both have at least one player
        # the one offender troops that stays on own territory is omitted
        while self.offender_troops>0 and self.defender_troops>0:
            self.attack()

        # determine winner based on troops left
        if self.offender_troops>0:
            offender_wins=True # for testing
            victories_for_troop_combos[self.offender_troops_orig][self.defender_troops_orig]+=1
        elif self.defender_troops>0:
            offender_wins=False # for testing

        #asign number of lost troops to a matrix based on original number of troops
        lost_troops_offender_this_round=self.offender_troops_orig - self.offender_troops
        lost_troops_defender_this_round=self.defender_troops_orig - self.defender_troops
        lost_troops_offender[self.offender_troops_orig][self.defender_troops_orig]+=lost_troops_offender_this_round
        lost_troops_defender[self.offender_troops_orig][self.defender_troops_orig]+=lost_troops_defender_this_round
        return self.offender_troops,self.defender_troops,offender_wins


# simulate battle many times with given nr of offender and defender troops
def simulate_given_battles(offender_troops, defender_troops, simulations=nr_of_simulations):
    for _ in range(simulations):
        one_battle=OneRisk(offender_troops=offender_troops,defender_troops=defender_troops)
        one_battle.calculate_outcome_of_one_battle()

# apply the above to exhaus all possible offender and defender combinations
def simulate_all_battles():
    for i in range(1,max_troops):
        for j in range(1,max_troops):
            simulate_given_battles(i,j)

    # raw numbers into expectations and percentages
    outcomes_of_dice_combos_percentages=100*outcomes_of_dice_combos/outcomes_of_dice_combos.sum(axis=2, keepdims=True)
    lost_troops_offender_expectation=pd.DataFrame(lost_troops_offender/nr_of_simulations)
    lost_troops_defender_expectation=pd.DataFrame(lost_troops_defender/nr_of_simulations)
    victories_for_troop_combos_percentages=pd.DataFrame(100*victories_for_troop_combos/nr_of_simulations)

    # formatting
    lost_troops_offender_expectation=lost_troops_offender_expectation.iloc[1:,1:].round(decimals=2)
    lost_troops_defender_expectation=lost_troops_defender_expectation.iloc[1:,1:].round(decimals=2)
    victories_for_troop_combos_percentages=victories_for_troop_combos_percentages.iloc[1:,1:].round(decimals=1)
    return outcomes_of_dice_combos_percentages,lost_troops_offender_expectation, lost_troops_defender_expectation,victories_for_troop_combos_percentages

# helper function to display dice percentages in a nicer format
def print_percentages(percentages):
    df=pd.DataFrame()
    for i in range(len(percentages)):
        for j in range(2):
            column=pd.DataFrame(percentages[i][j])
            column_name=f"{i+1}-{j+1}"
            column.columns=[column_name]
            df[column_name]=column
    rounded_df = df.round(decimals=1)
    print("Probabilities for a given number of wins in a single battle")
    print(" -> Column names indicate # offender dice - # defender dice")
    display(rounded_df)
